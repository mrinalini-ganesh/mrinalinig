1. the configmgmt3.yml script- name: Install cowsay and configure crontab
  hosts: linux1
  gather_facts: false
  tasks:
    - name: Install cowsay using apt
      apt:
        name: cowsay
        state: present

    - name: Add cowsay to crontab
      cron:
        name: cowsay every 5 minutes
        minute: "*/5"
        job: "/usr/games/cowsay Hi There!"

    - name: Verify cowsay installation
      shell: cowsay Hi There!
      register: cowsay_output
      changed_when: false
      failed_when: false

    - name: Purge cowsay package
      apt:
        name: cowsay
        state: absent

    - name: Reinstall cowsay package
      apt:
        name: cowsay
        state: present

    - name: Verify cowsay execution after reinstall
      shell: cowsay Hi There!
      register: cowsay_output_after_reinstall
      changed_when: false
      failed_when: false

- name: Print Ansible log
  hosts: localhost
  gather_facts: false
  tasks:
    - name: Read Ansible log file
      command: cat /home/ubuntu/ansible/ansible.log
      register: ansible_log_output
